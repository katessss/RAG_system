import chromadb
import os
from chromadb.config import Settings
from pathlib import Path
from logger_config import setup_logger
logger = setup_logger(__name__)

def save_to_chroma(chunks_with_vectors, collection_name, db_path):
    """
    Создает локальную базу данных и загружает в нее чанки с готовыми векторами.
    """
    db_dir = os.path.dirname(db_path)
    if db_dir:
        Path(db_dir).mkdir(parents=True, exist_ok=True)
        
    # Инициализируем клиента ChromaDB с указанием пути для хранения данных
    client = chromadb.PersistentClient(path=db_path)
    
    # Создаем или получаем коллекцию
    # отключаем встроенную модель эмбеддингов, так как используем свою
    collection = client.get_or_create_collection(name=collection_name, embedding_function=None)

    ids = []
    documents = []
    metadatas = []
    embeddings = []

    for i, item in enumerate(chunks_with_vectors):
        ids.append(f"id_{i}")
        documents.append(item["content"])
        # Метаданные (нужно превратить None в пустые строки или числа, Chroma не любит None)
        meta = {
            "page": item["metadata"].get("page") if item["metadata"].get("page") else 0,
            "context": item["metadata"].get("context", ""),
            "type": item.get("type") or item.get("metadata", {}).get("type") or "text"
        }
        metadatas.append(meta)
        embeddings.append(item["vectors"])

    # Загружаем всё в базу
    collection.upsert(
        ids=ids,
        metadatas=metadatas,
        documents=documents,
        embeddings=embeddings
    )
    
    logger.info(f"База данных готова! Загружено {len(ids)} чанков.")
    return collection
